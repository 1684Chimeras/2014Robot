392    //Scores the aspect ratio of a target. The aspect ratio of a target is it's sides over each other.
393    private double getAspectRatio(BinaryImage image, ParticleAnalysisReport report, int particleNumber, boolean vertical) throws NIVisionException
394    {
395        double rectLong, rectShort, aspectRatio = 0, idealAspectRatio;
396
397        //The longer sie of the particle
398        rectLong = NIVision.MeasureParticle(image.image, particleNumber, false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
399        //The shorter side of the particle
400        rectShort = NIVision.MeasureParticle(image.image, particleNumber, false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
401        idealAspectRatio = vertical ? (4.0/32) : (23.5/4);      //Vertical reflector 4" wide x 32" tall, horizontal 23.5" wide x 4" tall
402        
403        //Divide width by height to measure aspect ratio
404        if(report.boundingRectWidth > report.boundingRectHeight){
405            //particle is wider than it is tall, divide long by short
406            aspectRatio = getRatioScore((rectLong/rectShort)/idealAspectRatio);
407        }
408        return aspectRatio;
409    }

426    /*
427     * Determines if a target is rectangular or not.
428     * If the target is rectangular, than its # of pixels should be a constant % of its width * its height.
429     */
430    double getRectangularity(ParticleAnalysisReport report){
431        if(report.boundingRectWidth*report.boundingRectHeight !=0){
432                return 100*report.particleArea/(report.boundingRectWidth*report.boundingRectHeight);
433        } else {
434                return 0;
435        }       
436    }

438    /*
439     * Converts a ratio into a score.
440     * Desirably, the ratio should be at about 1. At 1, this function will return 100. At 0 or 2, this function will return 0. Else, it will interpolate.
441     */
442    double getRatioScore(double ratio)
443    {
444        return (Math.max(0, Math.min(100*(1-Math.abs(1-ratio)), 100)));
445    }
