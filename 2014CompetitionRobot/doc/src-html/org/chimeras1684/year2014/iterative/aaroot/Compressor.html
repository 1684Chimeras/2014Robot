012package org.chimeras1684.year2014.iterative.aaroot;
013
014import edu.wpi.first.wpilibj.DigitalInput;
015import edu.wpi.first.wpilibj.Relay;
016
017
018/**
019 *  Custom interface for compressors
020 *  because we're too cool to use WPI classes
021 * 
022 * @author Arhowk
023 */
024public class Compressor 
025{
026    Relay compressor;
027    DigitalInput pressureSwitch;
028    
029    /**
030     * Constructs a new singleton compressor because we can.
031     */
032    public Compressor()
033    {
034        compressor = new Relay(RobotMap.compressor);
035        pressureSwitch = new DigitalInput(RobotMap.pressureSwitch);
036    }
037    
038    /**
039     * Compresses the compressor if the relay is relaying
040     */
041    
042    public void compress()
043    {
044        if(pressureSwitch.get())
045        {
046            compressor.set(Relay.Value.kOff);
047        }else{
048            compressor.set(Relay.Value.kForward);
049        }    
050    }
051    
052    public void disable(){
053        compressor.set(Relay.Value.kOff);
054    }
055    
056}
