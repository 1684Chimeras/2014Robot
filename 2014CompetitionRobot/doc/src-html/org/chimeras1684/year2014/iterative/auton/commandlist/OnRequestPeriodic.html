005package org.chimeras1684.year2014.iterative.auton.commandlist;
006
007import org.chimeras1684.year2014.iterative.aaroot.Teleoperated;
008import org.chimeras1684.year2014.iterative.auton.root.OnRequest;
009
010/**
011 *
012 * @author Arhowk
013 */
014public class OnRequestPeriodic extends OnRequest{
015    
016    public static void runRequest(int request, double arg)
017    {
018        switch(request)
019        {
020            case intakeCustomSpeed:
021                intake.rollersAt(arg);
022                break;
023                
024            case intakeIn:
025                intake.frontRollerOut();
026                intake.rearRollerOut();
027                break;
028                
029            case intakeOut:
030                intake.frontRollerIn();
031                intake.rearRollerIn();
032                break;
033            
034            case frontIntakeIn:
035                intake.frontRollerOut();
036                break;
037                
038            case frontIntakeOut:
039                intake.frontRollerIn();
040                break;
041                
042            case rearIntakeIn:
043                intake.rearRollerOut();
044                break;
045                
046            case rearIntakeOut:
047                intake.rearRollerIn();
048                break;
049                
050            case intakeUp:
051                intake.frontIntakeRaise();
052                intake.rearIntakeRaise();
053                break;
054                
055            case intakeDown:
056                intake.frontIntakeLower();
057                intake.rearIntakeLower();
058                break;
059                
060            case frontIntakeUp:
061                intake.frontIntakeRaise();
062                break;
063                
064            case frontIntakeDown:
065                intake.frontIntakeLower();
066                break;
067                
068            case rearIntakeUp:
069                intake.rearIntakeRaise();
070                break;
071                
072            case rearIntakeDown:
073                intake.rearIntakeLower();
074                break;
075                
076            case kickerHold:
077                kicker.halt(arg  == 0 ? 180 : arg);
078                break;
079                
080            case kickerBackward:
081                kicker.set(arg == 0 ? -0.5 : -arg);
082                break;
083                
084            case kickerForward:
085                double ang = kicker.getKickerAngle();
086                if(ang > Teleoperated.hardstopStart && ang < Teleoperated.hardstopStop){
087                    kicker.set(-1);
088                }else{
089                    kicker.set(arg == 0 ? 0.5 : arg);
090                }
091                break;
092                
093            case compressorToggle:
094                compressor.compress();
095                break;
096                
097            case upperStructureEnable:
098                upperStructure.update();
099                break;
100                
101            case driveTurnLeftPID:
102                drive.rotate(arg == 0 ? -15 : -arg);
103                break;
104                
105            case driveTurnRightPID:
106                drive.rotate(arg == 0 ? 15 : arg);
107                break;
108                
109            case upperStructureManual:
110                upperStructure.manual(arg);
111                break;
112                
113            case driveTurnLeft:
114                drive.arcadeDrive(0, -1, true);
115                break;
116                
117            case driveTurnRight:
118                drive.arcadeDrive(0, 1, true);
119                break;
120                
121            case leftDrive:
122                drive.leftDrive(arg);
123                break;
124                
125                
126            case rightDrive:
127                drive.rightDrive(arg);
128                break;
129            
130            case driveForward:
131                drive.arcadeDrive(arg == 0 ? 1 : arg, arg < 0 ? -0.0 : -0.05,false);
132                break;
133                
134            case driveFeet:
135                drive.runPID();
136                break;
137                
138            case driveOff:
139                drive.arcadeDrive(0,0,false);
140                break;
141                
142            default:
143              //  System.out.println("[OnRequestPeriodic.java] Error : Set Request does not exist");
144        }
145    } //WHile false : stop!
146}
