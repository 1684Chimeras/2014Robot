274            upper = backingNativeMemory.getInt(4);
275        }
276
277        public void write() {
278            backingNativeMemory.setInt(0, lower);
279            backingNativeMemory.setInt(4, upper);
280        }
281
282        public int size() {
283            return 8;
284        }
285
286        /**
287         * Free the memory used by this range
288         */
289        public void free() {
290            release();
291        }
292
293        /**
294         * Create a new range with the specified upper and lower boundaries
295         * @param lower The lower limit
296         * @param upper The upper limit
297         */
298        public Range(final int lower, final int upper) {
299            allocateMemory();
300            set(lower, upper);
301        }
302
303        /**
304         * Set the upper and lower boundaries
305         * @param lower The lower limit
306         * @param upper The upper limit
307         */
308        public void set(final int lower, final int upper) {
309            if (lower > upper) {
310                throw new BoundaryException("Lower boundary is greater than upper");
311            }
312            this.lower = lower;
313            this.upper = upper;
314            write();
315        }
316
317        /**
318         * Get the lower boundary
319         * @return The lower boundary.
320         */
321        public int getLower() {
322            read();
323            return lower;
324        }
325
326        /**
327         * Get the upper boundary
328         * @return The upper boundary.
329         */
330        public int getUpper() {
331            read();
332            return upper;
333        }
